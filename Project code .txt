import streamlit as st
from pyswip import Prolog
import subprocess
from hospital import HospitalFinder  

class Patient:
    def __init__(self):
        self.symptoms = []

    def add_symptoms(self, symptoms):
        self.symptoms = symptoms

    def get_diagnosis(self):
        prolog = Prolog()
        prolog.consult("diagnosis.pl")  
        symptoms_list = "[" + ",".join(self.symptoms) + "]"
        result = list(prolog.query(f"diagnose({symptoms_list}, Diagnosis)"))
        if result:
            return result[0]['Diagnosis']
        return "general"

    def get_doctor(self, diagnosis):
        prolog = Prolog()
        prolog.consult("diagnosis.pl")  
        result = list(prolog.query(f"suggest_doctor({diagnosis}, Doctor)"))
        if result:
            doctor = result[0]['Doctor']
            if isinstance(doctor, bytes):
                doctor = doctor.decode("utf-8")
                return doctor
        return "General Physician"

    def get_lisp_advice(self, diagnosis):
        try:
            result = subprocess.run(
            ["C:\\Program Files\\Steel Bank Common Lisp\\sbcl.exe", "--script", "suggestion.lisp", diagnosis],
            capture_output=True, text=True,
            cwd="C:\\Users\\ayyss\\OneDrive\\Desktop\\AI"
)

            return result.stdout.strip()
        except Exception as e:
            return f"Error getting Lisp advice: {e}"

% Diagnosis rules

diagnose(Symptoms, flu) :-
    member(fever, Symptoms),
    member(cough, Symptoms),
    member(body_ache, Symptoms).

diagnose(Symptoms, heart_disease) :-
    member(chest_pain, Symptoms),
    member(shortness_of_breath, Symptoms).

diagnose(Symptoms, allergy) :-
    member(sneezing, Symptoms),
    member(itchy_eyes, Symptoms).

diagnose(Symptoms, migraine) :-
    member(headache, Symptoms),
    member(sensitivity_to_light, Symptoms).

diagnose(Symptoms, food_poisoning) :-
    member(nausea, Symptoms),
    member(vomiting, Symptoms),
    member(diarrhea, Symptoms).

diagnose(Symptoms, asthma) :-
    member(shortness_of_breath, Symptoms),
    member(wheezing, Symptoms),
    member(cough, Symptoms).

diagnose(Symptoms, depression) :-
    member(fatigue, Symptoms),
    member(no_motivation, Symptoms),
    member(lack_of_sleep, Symptoms).

diagnose(Symptoms, eczema) :-
    member(itchy_skin, Symptoms),

class HospitalFinder:
    def __init__(self):
        self.hospitals = {
            "General Physician": ["General Hospital", "City Care Clinic"],
            "Cardiologist": ["Heart Health Center", "Cardio Clinic"],
            "Allergist": ["Allergy and Immunology Center", "City Allergy Clinic"],
            "Neurologist": ["Day Institute", "NeuroCare Hospital"],
            "Gastroenterologist": ["Uno Health Center", "Digestive Care Clinic"],
            "Pulmonologist": ["Pulmonary Clinic", "Lung Health Center"],
            "Psychiatrist": ["Mental Health Institute", "City Psychiatry Clinic"],
            "Dermatologist": ["Skin Health Center", "DermCare Clinic"]
        }

    def find_hospitals(self, specialist):
        return self.hospitals.get(specialist, [])
